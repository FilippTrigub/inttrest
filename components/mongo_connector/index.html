    MongoDB Connector¶

    The MongoDB connector provides a simple, clean interface for interacting with MongoDB Atlas. It handles connections, data insertion, querying, and error management.

    üîó Overview¶

    graph LR
    APP[Application] --> CONNECTOR[MongoConnector]
    CONNECTOR --> CLIENT[MongoDB Client]
    CLIENT --> ATLAS[(MongoDB Atlas)]

    CONNECTOR --> VALIDATION[Data Validation]
    CONNECTOR --> INDEXING[Index Management]
    CONNECTOR --> ERROR[Error Handling]

    style CONNECTOR fill:#e8f5e8
    style ATLAS fill:#e3f2fd

    Querying Events¶

    # Find events by city
paris_events = mongo.find_events(city="Paris", limit=10)
print(f"Found {len(paris_events)} events in Paris")

# Find events by category
tech_events = mongo.find_events(category="Technology", limit=5)

# Find all events (no filters)
all_events = mongo.find_events(limit=50)

# Count total events
total = mongo.count_events()
print(f"Total events in database: {total}")

    üîÑ Data Flow¶

    sequenceDiagram
    participant App as Application
    participant Connector as MongoConnector
    participant Client as MongoDB Client
    participant Atlas as MongoDB Atlas

    App->>Connector: insert_event(data)
    Connector->>Connector: Validate data

    alt Data is valid
        Connector->>Connector: Add timestamp
        Connector->>Client: insert_one(document)
        Client->>Atlas: Insert document
        Atlas-->>Client: Acknowledge
        Client-->>Connector: InsertResult
        Connector-->>App: True (success)
    else Data is invalid
        Connector-->>App: False (failure)
    end

    Note over App,Atlas: Error handling at each step

    Automatic Index Creation¶

    The connector automatically creates these indexes for optimal performance:

    # Primary index (unique)
collection.create_index("event_id", unique=True)

# Geospatial queries
collection.create_index([("latitude", 1), ("longitude", 1)])

# Common search fields
collection.create_index("city")
collection.create_index("category") 
collection.create_index("start_datetime")
collection.create_index("platform")

# Compound index for complex queries
collection.create_index([
    ("city", 1),
    ("start_datetime", 1), 
    ("category", 1)
])

    Network Errors¶

    # Automatic retry with exponential backoff
from time import sleep
import random

def robust_insert(mongo, event, max_retries=3):
    for attempt in range(max_retries):
        try:
            return mongo.insert_event(event)
        except Exception as e:
            if attempt == max_retries - 1:
                raise e
            sleep(2 ** attempt + random.uniform(0, 1))
    return False

    Connection Health Check¶

    def health_check(mongo):
    """Check if MongoDB connection is healthy"""
    try:
        # Simple ping test
        mongo.client.admin.command('ping')

        # Test basic operations
        test_count = mongo.count_events()

        return {
            "status": "healthy",
            "total_events": test_count,
            "connection": "active"
        }
    except Exception as e:
        return {
            "status": "unhealthy", 
            "error": str(e),
            "connection": "failed"
        }

    Performance Metrics¶

    import time

def timed_operation(func, *args, **kwargs):
    """Measure operation execution time"""
    start = time.time()
    result = func(*args, **kwargs)
    duration = time.time() - start

    print(f"Operation took {duration:.3f} seconds")
    return result

# Usage
inserted = timed_operation(mongo.insert_many_events, events)

    import unittest
from mongo_connector import MongoConnector

class TestMongoConnector(unittest.TestCase):
    def setUp(self):
        self.mongo = MongoConnector("test_password")

    def test_event_insertion(self):
        event = {
            "event_id": "test_001",
            "title": "Test Event",
            "latitude": 48.8566,
            "longitude": 2.3522
        }

        result = self.mongo.insert_event(event)
        self.assertTrue(result)

    def test_event_retrieval(self):
        events = self.mongo.find_events(limit=1)
        self.assertIsInstance(events, list)

    def tearDown(self):
        self.mongo.close()
