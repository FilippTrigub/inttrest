    Geocoding Service¶

    The geocoding service converts addresses and location names into precise latitude and longitude coordinates using the free OpenStreetMap Nominatim API.

    🗺️ Overview¶

    graph LR
    INPUT[Address String] --> GEOCODER[Geocoding Service]
    GEOCODER --> NOMINATIM[Nominatim API]
    NOMINATIM --> PARSER[Response Parser]
    PARSER --> COORDS[Lat/Lng Coordinates]

    GEOCODER --> CACHE[Local Cache]
    CACHE --> COORDS

    🔧 Implementation¶

    Core Geocoder Class¶

    import requests
import time
from typing import Tuple, Optional

class FreeGeocoder:
    """Free geocoding using OpenStreetMap Nominatim"""

    def __init__(self):
        self.base_url = "https://nominatim.openstreetmap.org/search"
        self.session = requests.Session()
        self.session.headers.update({
            'User-Agent': 'EventApp/1.0 (your-email@example.com)'
        })
        self.cache = {}  # Simple in-memory cache

    def geocode(self, address: str) -> Optional[Tuple[float, float]]:
        """Convert address to latitude, longitude coordinates"""

        if not address or address.strip() == '':
            return None

        # Check cache first
        cache_key = address.lower().strip()
        if cache_key in self.cache:
            return self.cache[cache_key]

        # Rate limiting - Nominatim allows 1 request per second
        time.sleep(1)

        params = {
            'q': address,
            'format': 'json',
            'limit': 1,
            'addressdetails': 1
        }

        try:
            response = self.session.get(
                self.base_url, 
                params=params, 
                timeout=5
            )
            response.raise_for_status()

            data = response.json()

            if data and len(data) > 0:
                result = data[0]
                lat = float(result['lat'])
                lon = float(result['lon'])

                coords = (lat, lon)

                # Cache the result
                self.cache[cache_key] = coords

                print(f"✅ Geocoded '{address}' → {lat:.4f}, {lon:.4f}")
                return coords
            else:
                print(f"❌ No results for '{address}'")
                return None

        except Exception as e:
            print(f"❌ Geocoding error for '{address}': {e}")
            return None

    Integration with Event Processing¶

    def process_event_location(event_data):
    """Process event and add coordinates if missing"""

    # Check if coordinates already exist
    lat = event_data.get('latitude', 0)
    lng = event_data.get('longitude', 0)

    if lat != 0 and lng != 0:
        print(f"✅ Event already has coordinates: {lat}, {lng}")
        return event_data

    # Try to get location from various fields
    location_sources = [
        event_data.get('address', ''),
        event_data.get('venue_name', ''),
        event_data.get('location', ''),
        event_data.get('city', '')
    ]

    # Find first non-empty location
    location = next((loc for loc in location_sources if loc.strip()), None)

    if location:
        coords = geocoder.geocode(location)
        if coords:
            event_data['latitude'] = coords[0]
            event_data['longitude'] = coords[1]
            event_data['geocoded_from'] = location
            print(f"🗺️ Added coordinates from '{location}'")
        else:
            print(f"⚠️ Could not geocode: {location}")
            # Set default coordinates (Paris center)
            event_data['latitude'] = 48.8566
            event_data['longitude'] = 2.3522
            event_data['geocoded_from'] = 'default_paris'

    return event_data

    🔄 Geocoding Workflow¶

    sequenceDiagram
    participant Event as Event Data
    participant Processor as Event Processor
    participant Cache as Local Cache
    participant Geocoder as Geocoder
    participant Nominatim as Nominatim API

    Event->>Processor: Raw event data
    Processor->>Processor: Check existing coordinates

    alt Has coordinates
        Processor-->>Event: Return with existing coords
    else No coordinates
        Processor->>Processor: Extract location string
        Processor->>Cache: Check cache

        alt Cache hit
            Cache-->>Processor: Return cached coords
        else Cache miss
            Processor->>Geocoder: geocode(address)
            Geocoder->>Nominatim: HTTP request
            Nominatim-->>Geocoder: JSON response
            Geocoder->>Cache: Store result
            Geocoder-->>Processor: Coordinates
        end

        Processor->>Event: Add coordinates to event
    end

    🚀 Performance Optimization¶

    Caching Strategy¶

    class CachedGeocoder(FreeGeocoder):
    """Enhanced geocoder with persistent caching"""

    def __init__(self, cache_file="geocode_cache.json"):
        super().__init__()
        self.cache_file = cache_file
        self.load_cache()

    def load_cache(self):
        """Load cache from file"""
        try:
            with open(self.cache_file, 'r') as f:
                import json
                self.cache = json.load(f)
            print(f"📂 Loaded {len(self.cache)} cached locations")
        except FileNotFoundError:
            self.cache = {}

    def save_cache(self):
        """Save cache to file"""
        try:
            with open(self.cache_file, 'w') as f:
                import json
                json.dump(self.cache, f, indent=2)
            print(f"💾 Saved {len(self.cache)} locations to cache")
        except Exception as e:
            print(f"❌ Cache save failed: {e}")

    def geocode(self, address: str) -> Optional[Tuple[float, float]]:
        """Geocode with persistent caching"""
        result = super().geocode(address)

        # Save cache after each geocoding operation
        if result and len(self.cache) % 10 == 0:  # Save every 10 new entries
            self.save_cache()

        return result

    Batch Geocoding¶

    def batch_geocode(addresses, batch_size=10):
    """Geocode multiple addresses with rate limiting"""

    geocoder = CachedGeocoder()
    results = {}

    print(f"🔄 Geocoding {len(addresses)} addresses...")

    for i, address in enumerate(addresses):
        if i > 0 and i % batch_size == 0:
            print(f"📊 Progress: {i}/{len(addresses)} completed")
            # Longer pause every batch_size requests
            time.sleep(5)

        coords = geocoder.geocode(address)
        results[address] = coords

        # Show progress for long lists
        if len(addresses) > 20 and i % 5 == 0:
            print(f"⏳ {i+1}/{len(addresses)} processed...")

    # Save final cache
    geocoder.save_cache()

    return results

    🌐 Alternative Geocoding Services¶

    Google Geocoding API (Premium)¶

    class GoogleGeocoder:
    """Google Geocoding API for higher accuracy"""

    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://maps.googleapis.com/maps/api/geocode/json"

    def geocode(self, address: str) -> Optional[Tuple[float, float]]:
        """Geocode using Google Maps API"""

        params = {
            'address': address,
            'key': self.api_key
        }

        try:
            response = requests.get(self.base_url, params=params)
            data = response.json()

            if data['status'] == 'OK' and data['results']:
                location = data['results'][0]['geometry']['location']
                return (location['lat'], location['lng'])

        except Exception as e:
            print(f"Google geocoding error: {e}")

        return None

    Fallback Strategy¶

    def robust_geocode(address: str) -> Optional[Tuple[float, float]]:
    """Try multiple geocoding services as fallback"""

    # Try Nominatim first (free)
    coords = geocoder.geocode(address)
    if coords:
        return coords

    # If Google API key is available, try Google
    google_api_key = os.getenv('GOOGLE_GEOCODING_API_KEY')
    if google_api_key:
        google_geocoder = GoogleGeocoder(google_api_key)
        coords = google_geocoder.geocode(address)
        if coords:
            return coords

    # If all fail, try to extract city and use city center
    if 'paris' in address.lower():
        return (48.8566, 2.3522)  # Paris center
    elif 'london' in address.lower():
        return (51.5074, -0.1278)  # London center

    return None

    Geocoding Success Rate¶

    def analyze_geocoding_quality(addresses):
    """Analyze geocoding success rate and quality"""

    results = {
        'total': len(addresses),
        'successful': 0,
        'failed': 0,
        'paris_area': 0,
        'outside_paris': 0
    }

    for address in addresses:
        coords = geocoder.geocode(address)

        if coords:
            results['successful'] += 1
            lat, lng = coords

            # Check if in Paris area (rough bounds)
            if 48.8 <= lat <= 48.9 and 2.2 <= lng <= 2.5:
                results['paris_area'] += 1
            else:
                results['outside_paris'] += 1
        else:
            results['failed'] += 1

    # Calculate percentages
    results['success_rate'] = (results['successful'] / results['total']) * 100

    return results
