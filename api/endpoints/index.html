    API Reference¶

    This document provides comprehensive API documentation for the Event Discovery application.

    üìã Overview¶

    The Event Discovery API provides endpoints for managing events, geocoding locations, and integrating with external data sources through MCP (Model Context Protocol) servers.

    üìç Event Endpoints¶

    GET /api/v1/events¶

    Retrieve events from the database with optional filtering.

    Request Parameters:

    Parameter Type Required Description
    limit integer No Maximum number of events to return (default: 100)
    offset integer No Number of events to skip (default: 0)
    category string No Filter by event category
    location string No Filter by location name
    date_from string No Filter events from date (ISO format)
    date_to string No Filter events until date (ISO format)
    source string No Filter by data source (eventbrite, meetup, etc.)
    has_coordinates boolean No Filter events with/without coordinates

    Example Request: 

    curl -X GET "http://localhost:8000/api/v1/events?limit=50&category=technology&has_coordinates=true" \
  -H "Content-Type: application/json"

    Response: 

    {
  "events": [
    {
      "id": "evt_001",
      "title": "Tech Meetup Paris",
      "description": "Monthly technology meetup in Paris",
      "location": "15 rue de Milan, Paris",
      "coordinates": {
        "latitude": 48.8786,
        "longitude": 2.3273
      },
      "date": "2024-02-15T18:00:00Z",
      "category": "technology",
      "source": "eventbrite",
      "image_url": "https://example.com/event-image.jpg",
      "url": "https://eventbrite.com/event/123",
      "created_at": "2024-01-15T10:30:00Z",
      "updated_at": "2024-01-15T10:30:00Z"
    }
  ],
  "total": 1,
  "limit": 50,
  "offset": 0,
  "has_more": false
}

    Status Codes: - 200: Success - 400: Invalid parameters - 500: Server error

    üêç Python Client Examples¶

    Basic Usage¶

    import requests
import json

class EventDiscoveryClient:
    def __init__(self, base_url="http://localhost:8000"):
        self.base_url = base_url
        self.session = requests.Session()

    def get_events(self, **filters):
        """Get events with optional filters"""
        response = self.session.get(
            f"{self.base_url}/api/v1/events",
            params=filters
        )
        response.raise_for_status()
        return response.json()

    def create_event(self, event_data, auto_geocode=True):
        """Create a new event"""
        payload = {
            "events": [event_data],
            "auto_geocode": auto_geocode
        }

        response = self.session.post(
            f"{self.base_url}/api/v1/events",
            json=payload
        )
        response.raise_for_status()
        return response.json()

    def geocode_addresses(self, addresses):
        """Geocode multiple addresses"""
        payload = {"addresses": addresses}

        response = self.session.post(
            f"{self.base_url}/api/v1/geocode",
            json=payload
        )
        response.raise_for_status()
        return response.json()

    def sync_with_mcp(self, sources, location=None, category=None):
        """Trigger MCP synchronization"""
        payload = {
            "sources": sources,
            "auto_geocode": True
        }

        if location:
            payload["location"] = location
        if category:
            payload["category"] = category

        response = self.session.post(
            f"{self.base_url}/api/v1/mcp/sync",
            json=payload
        )
        response.raise_for_status()
        return response.json()

# Example usage
client = EventDiscoveryClient()

# Get technology events in Paris
events = client.get_events(
    category="technology",
    location="Paris",
    limit=20
)

# Create a new event
new_event = {
    "id": "custom_001",
    "title": "Python Workshop",
    "location": "Station F, Paris",
    "date": "2024-02-20T14:00:00Z",
    "category": "technology"
}

result = client.create_event(new_event)
print(f"Created event: {result}")

# Geocode addresses
addresses = ["15 rue de Milan, Paris", "Times Square, NYC"]
coords = client.geocode_addresses(addresses)
print(f"Geocoding results: {coords}")

# Sync with Eventbrite
sync_result = client.sync_with_mcp(
    sources=["eventbrite"],
    location="Paris, France",
    category="technology"
)
print(f"Sync started: {sync_result}")

    Advanced Examples¶

    # Bulk event processing
class BulkEventProcessor:
    def __init__(self, client):
        self.client = client

    def process_csv_events(self, csv_file_path):
        """Process events from CSV file"""
        import csv

        events = []
        with open(csv_file_path, 'r') as file:
            reader = csv.DictReader(file)
            for row in reader:
                event = {
                    "id": row["id"],
                    "title": row["title"],
                    "location": row["location"],
                    "date": row["date"],
                    "category": row.get("category", "other")
                }
                events.append(event)

        # Process in batches of 50
        batch_size = 50
        results = []

        for i in range(0, len(events), batch_size):
            batch = events[i:i + batch_size]

            payload = {
                "events": batch,
                "auto_geocode": True
            }

            result = self.client.session.post(
                f"{self.client.base_url}/api/v1/events",
                json=payload
            )
            result.raise_for_status()
            results.append(result.json())

        return results

    def export_events_to_csv(self, output_file, **filters):
        """Export events to CSV"""
        import csv

        # Get all events with filters
        events_response = self.client.get_events(**filters)
        events = events_response["events"]

        # Write to CSV
        with open(output_file, 'w', newline='') as file:
            if events:
                fieldnames = events[0].keys()
                writer = csv.DictWriter(file, fieldnames=fieldnames)
                writer.writeheader()
                writer.writerows(events)

        return len(events)

# Usage
processor = BulkEventProcessor(client)

# Import events from CSV
results = processor.process_csv_events("events.csv")
print(f"Imported {sum(r['inserted_count'] for r in results)} events")

# Export technology events to CSV
count = processor.export_events_to_csv(
    "tech_events.csv",
    category="technology"
)
print(f"Exported {count} events")

    Basic Fetch API¶

    class EventDiscoveryAPI {
    constructor(baseUrl = 'http://localhost:8000') {
        this.baseUrl = baseUrl;
    }

    async getEvents(filters = {}) {
        const params = new URLSearchParams(filters);
        const response = await fetch(`${this.baseUrl}/api/v1/events?${params}`);

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        return await response.json();
    }

    async createEvent(eventData, autoGeocode = true) {
        const response = await fetch(`${this.baseUrl}/api/v1/events`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({
                events: [eventData],
                auto_geocode: autoGeocode
            })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        return await response.json();
    }

    async geocodeAddresses(addresses) {
        const response = await fetch(`${this.baseUrl}/api/v1/geocode`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ addresses })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        return await response.json();
    }
}

// Usage
const api = new EventDiscoveryAPI();

// Get events and display on map
async function loadEventsOnMap() {
    try {
        const response = await api.getEvents({
            limit: 100,
            has_coordinates: true
        });

        response.events.forEach(event => {
            if (event.coordinates) {
                addMarkerToMap(
                    event.coordinates.latitude,
                    event.coordinates.longitude,
                    event.title,
                    event.description
                );
            }
        });

    } catch (error) {
        console.error('Failed to load events:', error);
    }
}

    import pytest
import requests
from unittest.mock import patch

class TestEventAPI:

        return EventDiscoveryClient("http://localhost:8000")

        assert "events" in events
        assert "total" in events
        assert len(events["events"]) <= 10

    def test_get_events_with_filters(self, api_client):
        """Test event retrieval with filters"""
        events = api_client.get_events(
            category="technology",
            has_coordinates=True
        )

        for event in events["events"]:
            assert event["category"] == "technology"
            assert "coordinates" in event

    @patch('requests.Session.post')

    def test_geocoding_multiple_addresses(self, api_client):

    class TestEventAPIIntegration:
